    void calcular() {
        while (true) {

            double maiorn = 0;
            int col = -1;
            // descobrir a coluna do maior negativo e o maior negativo
            for (int i = 0; i < coluna; i++) {

                if (z[i] < maiorn) {
                    maiorn = z[i];
                    col = i;

                }
            }
            if (maiorn == 0) {
                break;
            }

            //Pegar os números da coluna encontrada e usa-los para dividir com o ultimo valor do vetor [coluna-1]
            // mrpd = Menor resultado positivo da divisão
            // pivo = numero pivo 
            // novalpivo = nova linha pivo 
            // linha = linha do pivo
            double mrpd = 999999;
            int linha = -1;
            for (int i = 0; i < f; i++) {
                double div = (restricao[i][coluna - 1] / restricao[i][col]);
                if (div < mrpd) {
                    if (div > 0) {
                        mrpd = restricao[i][coluna - 1] / restricao[i][col];
                        linha = i;
                    }
                }
            }

            double pivo = restricao[linha][col];

            //definir a nova linha pivo 
            novalpivo = new double[coluna];
            for (int i = 0; i < coluna; i++) {
                novalpivo[i] = restricao[linha][i] / pivo;
            }
            // novalpivo multiplicado por (-1*z[col]) o resultado somado a linha correspondente ao z[]

            double e = z[col];
            for (int i = 0; i < coluna; i++) {
                z[i] = novalpivo[i] * (-1 * e) + z[i];

            }

            /* Correndo as restrições SE ela forem DIFERENTES da linha pivo, multiplicando pelo inverso do numero correspondente ao COL, somando com o seu
	estado anterior e substituindo */
            for (int i = 0; i < f; i++) {
                e = restricao[i][col];
                for (int j = 0; j < coluna; j++) {

                    if (linha != i) {
                        restricao[i][j] = novalpivo[j] * (-1 * e) + restricao[i][j];
                    } else {
                        restricao[i][j] = novalpivo[j];
                    }
                }
            }
        }
    }

    double[][] variavel;

    void pegarResultado(JPanel pnlSaida) {
        variavel = new double[coluna - 1][2];

        int c0, c1;
        for (int i = 0; i < coluna - 1; i++) { // colunas
            c1 = 0;
            c0 = 0;
            int linha = -2;
            if (z[i] == 1) {
                c1++;
                linha = -1;
            } else if (z[i] == 0) {
                c0++;
            }
            for (int j = 0; j < f; j++) {//linhas
                if (restricao[j][i] == 1) {
                    c1++;
                    linha = j;
                } else if (restricao[j][i] == 0) {
                    c0++;
                }
            }
            if ((c1 + c0) == (f + 1)) {

                if (linha == -1) {

                    variavel[i][0] = 1;
                    variavel[i][1] = z[coluna - 1];

                } else if (linha > -1) {

                    variavel[i][0] = 1;
                    variavel[i][1] = restricao[linha][coluna - 1];
                } else {
                    variavel[i][0] = 0;
                    variavel[i][1] = 0;
                }
            }

        }

        String Resultado = "";
        for (int i = 0; i < coluna - 1; i++) {

            if (variavel[i][0] == 0) {
                Resultado += ("Variavel não básica - ");
            } else {
                if (i == 0) {
                    Resultado += ("Solução - ");
                } else {

                    Resultado += ("Variavel básica - ");
                }
            }
            if (i == 0) {
                Resultado += ("Z - ");
            } else if (i <= x) {
                Resultado += ("x" + i + " - ");
            } else {
                Resultado += ("f" + (i - x) + " - ");
            }

            Resultado += (variavel[i][1]);
            Resultado += ("\n");
        }

        TrocarPainel(pnlSaida, pnlResultado);
        txtResultado.setText(Resultado);
    }

        jPanel1.add(pnlEntrada);
        this.pack();
        this.setSize(665, 512);

    }
